{
  void fly(){
    print('I m flying');
  }
}

}

{

  int age =0;

  Animal(this.name, this.age, this.weight);
  
  void eat()
  {
    print('$name is Eating');
    weight = weight + 0.2;
  }
  void sleep()
  {
    print('$name + is Sleeping');
  }
  
}
